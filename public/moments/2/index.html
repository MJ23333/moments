在 MusicFox 中加入快进功能时，使用了以下代码：

func (p *Player) Speedup() {
	speaker.Lock()
	pos := p.seeker.Position()
	pos += p.SampleRate.N(10 * time.Second)
	err := p.seeker.Seek(pos)
	if err == nil {
		p.timechange += 10 * time.Second
	}
	speaker.Unlock()
}

if _, ok := d.source.reader.(io.Seeker); !ok {
